void jjack(int arr[], int size);
int main(void) {

		printf("입력: ");

void hol(int arr[], int size) {
	printf("홀수 출력: ");
	for (int i = 0; i < size; i++) {
		if (arr[i] % 2 != 0)
			printf("%d, ", arr[i]);
	}

}

void jjack(int arr[], int size) {
	printf("짝수 출력: ");
	for (int i = 0; i < size; i++) {
		if (arr[i] % 2 == 0)
			printf("%d, ", arr[i]);
	}

}
