#define SIZE 10
void _sum(int* arr, int* sum) {
	for (int i = 0; i < SIZE; i++)
		*sum += *(arr + i);
	printf("√—«’ = %d\n", *sum);
}

	*avg = (double)*sum / (double)SIZE;
	printf("∆Ú±’ = %lf\n", *avg);
}

			*min = *(arr + i);
			printf("√÷º“ = %d\n", *min);
}
void _max(int* arr, int* max) {
	for (int i = 0; i < SIZE; i++)
		if (*(arr + i) > * max)
			* max = *(arr + i);
	printf("√÷¥Î = %d", *max);
}
